\begin{document}
\begin{tikzpicture}[scale=2]

% Define base radius
\def\R{4}

% Clock face outer circle
\draw[line width=2pt] (0,0) circle (\R);
\draw[line width=0.5pt] (0,0) circle (0.925*\R);

% Hour markers and canton positions
\foreach \hour/\outerCanton/\innerCanton in {
    1/\cantonOne/\cantonTwo,
    2/\cantonThree/\cantonFour,
    3/\cantonFive/\cantonSix,
    4/\cantonSeven/\cantonEight,
    5/\cantonNine/\cantonTen,
    6/\cantonEleven/\cantonTwelve,
    7/\cantonThirteen/\cantonFourteen,
    8/\cantonFifteen/\cantonSixteen,
    9/\cantonSeventeen/\cantonEighteen,
    10/\cantonNineteen/\cantonTwenty,
    11/\cantonTwentyOne/\cantonTwentyTwo,
    12/\cantonTwentyThree/\cantonTwentyFour
} {
    % Calculate angle (12 o'clock is at 90 degrees)
    \pgfmathsetmacro{\angle}{90 - \hour * 30}
    
    % Hour tick marks
    \draw[line width=1.5pt] (\angle:0.875*\R) -- (\angle:0.925*\R);
    
    % Hour numbers
    \node at (\angle:0.65*\R) {\Large\textsf{\hour}};
    
    % Outer canton coat of arms
    \draw[thick, fill=white] (\angle:0.825*\R) ++(-0.0875*\R,-0.0875*\R) rectangle ++(0.175*\R,0.175*\R);
    \begin{scope}[shift={(\angle:0.825*\R)}]
        \outerCanton
    \end{scope}
    
    % Inner canton coat of arms - between hours
    \pgfmathsetmacro{\innerangle}{\angle - 15}
    \draw[thick, fill=white] (\innerangle:0.45*\R) ++(-0.075*\R,-0.075*\R) rectangle ++(0.15*\R,0.15*\R);
    \begin{scope}[shift={(\innerangle:0.45*\R)}]
        \innerCanton
    \end{scope}
}

% Minute markers
\foreach \minute in {0,6,12,...,354} {
    \pgfmathsetmacro{\angle}{90 - \minute}
    \pgfmathsetmacro{\modThirty}{int(mod(\minute, 30))}
    
    \ifnum\modThirty=0
        % Skip - this is an hour position (already drawn as thick line)
    \else
        % Not an hour position - check if it's a 5-minute interval
        \pgfmathsetmacro{\modThirtySub}{int(mod(\minute, 30))}
        \ifnum\modThirtySub=0
            % This is never reached since we already excluded mod 30 = 0
        \else
            % Check actual 5-minute intervals (every 30 degrees = 5 minutes)
            % Since input is every 6 degrees, 5 min = 30 deg
            % Just draw all non-hour markers the same for now
            \draw[line width=0.5pt] (\angle:0.9*\R) -- (\angle:0.925*\R);
        \fi
    \fi
}

% Center canton positions (two remaining cantons)
\draw[thick, fill=white] (-0.125*\R, 0) ++(-0.0875*\R,-0.0875*\R) rectangle ++(0.175*\R,0.175*\R);
\begin{scope}[shift={(-0.125*\R, 0)}]
    \cantonTwentyFive
\end{scope}

\draw[thick, fill=white] (0.125*\R, 0) ++(-0.0875*\R,-0.0875*\R) rectangle ++(0.175*\R,0.175*\R);
\begin{scope}[shift={(0.125*\R, 0)}]
    \cantonTwentySix
\end{scope}

% Clock hands
% Hour hand (pointing to 10)
\draw[line width=3pt, line cap=round] (0,0) -- (150:0.375*\R);

% Minute hand (pointing to 2)
\draw[line width=2pt, line cap=round] (0,0) -- (-60:0.575*\R);

% Second hand (pointing to 6)
\draw[line width=1pt, red, line cap=round] (0,0) -- (-90:0.625*\R);

% Center cap
\fill (0,0) circle (0.0375*\R);
\draw[line width=1pt] (0,0) circle (0.0375*\R);

\end{tikzpicture}
\end{document}
